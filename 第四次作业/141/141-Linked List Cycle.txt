public class Solution {
   public boolean hasCycle(ListNode head) {
    if(head == null || head.next == null)
           return false;
        
         ListNode Faster = head, Slower = head;
        
        while(Faster.next!=null && Faster.next.next!=null){
            Slower = Slower.next;
         Faster = Faster.next.next;
           
            if(Faster == Slower)
                 return true;
        }
        return false;
     }
}
