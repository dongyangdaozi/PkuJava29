public class Solution {
    public int removeDuplicates(int[] nums) {
        int len=nums.length;
        int minus=0;
        if(len==1)return 1;
        for(int i=0;i<len-1-minus;){
            if(nums[i]==nums[i+1])
            {
                for(int j=i;j<len-1-minus;j++)
                {
                    nums[j]=nums[j+1];
                }
                minus++;
            }
            if(nums[i]!=nums[i+1])i++;
        }
        return len-minus;
    }
}
