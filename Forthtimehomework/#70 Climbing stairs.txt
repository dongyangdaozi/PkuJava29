public class Solution {
    public int climbStairs(int n) {
        //if(n==1)return 1;
        //if(n==2)return 2;
        //return climbStairs(n-1)+climbStairs(n-2);
        if(n==0||n==1||n==2)return n;
        int[] temp=new int[n+1];
        temp[0]=0;temp[1]=1;temp[2]=2;
        for(int i=3;i<=n;i++){
            temp[i]=temp[i-1]+temp[i-2];
        }
        return temp[n];
    }
}
